#include <stdio.h>
int main(void)
{
	int i,j,c,a[10];
	for(i=1;i<10;i++)
	{
		scanf("%d",&a[i]);
	}
	a[0]=0;
	for(c=2;c<10;c++)
	{
		a[0]=a[c];j=c-1;
		while(a[0]<a[j])
		{
			a[j+1]=a[j];
			j--;
		}
		a[j+1]=a[0];
	}
	for(i=1;i<10;i++)
	{
		printf("%d",a[i]);
	}
	return 0;
}
